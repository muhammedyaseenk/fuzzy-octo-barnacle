$flutterBin = "D:\auram_sharahiya\flutter\bin"
$oldPath = [Environment]::GetEnvironmentVariable('Path','User')

if ($oldPath -notlike "*$flutterBin*") {
    [Environment]::SetEnvironmentVariable('Path', "$oldPath;$flutterBin", 'User')
    Write-Output "Added Flutter to PATH."
} else {
    Write-Output "Flutter already on PATH."
}
